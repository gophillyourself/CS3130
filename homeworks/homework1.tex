    Find $gcd(213486, 5423)$

    \begin{linenumbers}
      213486 / 5243 = 39 R 1989 \\
      213486 = 5423 * 39 + 1989 \\
      213486 / 1989 = 107 R 663 \\
      213486 = 1989 * 107 + 663 \\
      213486 / 663 = 332 R 0 

    \end{linenumbers}

    $\therefore gcd(213486,5423) = 663 $
